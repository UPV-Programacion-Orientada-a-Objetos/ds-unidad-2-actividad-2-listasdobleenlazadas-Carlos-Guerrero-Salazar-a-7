14/**
15 * @brief Parsea una línea del serial y crea la trama correspondiente
16 * @param linea Cadena recibida del serial (ej. "L,A" o "M,5")
17 * @return Puntero a TramaBase (TramaLoad o TramaMap) o nullptr si hay error
18 */
19TramaBase* parsearTrama(const char* linea) {
20 if (!linea || linea[0] == '\0') {
21 return nullptr;
22 }
23
24 // Verificar formato mínimo: "X,Y"
25 if (std::strlen(linea) < 3 || linea[1] != ',') {
26 std::cerr << "[ERROR] Trama mal formada: " << linea << std::endl;
27 return nullptr;
28 }
29
30 char tipo = linea[0];
31
32 if (tipo == 'L') {
33 // Trama de carga: L,X donde X es un carácter
34 char caracter = linea[2];
35 return new TramaLoad(caracter);
36 }
37 else if (tipo == 'M') {
38 // Trama de mapeo: M,N donde N es un entero
39 int rotacion = std::atoi(&linea[2]);
40 return new TramaMap(rotacion);
41 }
42 else {
43 std::cerr << "[ERROR] Tipo de trama desconocido: " << tipo << std::endl;
44 return nullptr;
45 }
46}
